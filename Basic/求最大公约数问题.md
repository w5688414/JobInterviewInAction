# problem
> 总时间限制：1000ms  内存限制：65536KB
     给定两个正整数，求它们的最大公约数。
# codes
```
#include <iostream>
using namespace std;

int gcd(int m, int n){
    if(m%n==0){
        return n;
    }
    else return gcd(n,m%n);
}

int main(int argc, const char * argv[]) {
    // insert code here...
    int m,n;
    cin>>m>>n;
    int result=gcd(m,n);
    printf("%d",result);
    return 0;
}

```
